  <script>
    // Background music logic
    const music = document.getElementById("bgMusic");
    const btn = document.getElementById("toggleMusic");
    btn.addEventListener("click", () => {
      if (music.paused) {
        music.play();
        btn.textContent = "‚è∏ Pause Music";
      } else {
        music.pause();
        btn.textContent = "üéµ Play Music";
      }
    });

    function openGame(game) {
      const gameArea = document.getElementById("gameArea");
      gameArea.innerHTML = "";

      if (game === "puzzle") loadPuzzleGame(gameArea);
      else if (game === "rhythm") loadRhythmGame(gameArea);
      else if (game === "clouds") loadCloudGame(gameArea);
    }

    // üß© Calm Puzzle
    function loadPuzzleGame(container) {
      container.innerHTML = "<h3>üß© Calm Puzzle</h3><canvas id='puzzleCanvas' width='300' height='300'></canvas><p id='resultMessage'></p>";
      const ctx = document.getElementById("puzzleCanvas").getContext("2d");
      const img = new Image();
      img.src = "assets/images/nature.jpg"; // peaceful image
      img.onload = () => {
        ctx.drawImage(img, 0, 0, 300, 300);
        setTimeout(() => ctx.drawImage(img, 150, 150, 150, 150, 0, 0, 150, 150), 500);
      };
      setTimeout(() => showResult("Puzzle complete! üåø"), 3000);
    }

    // üéµ Rhythm Tap
    function loadRhythmGame(container) {
      container.innerHTML = "<h3>üéµ Rhythm Tap</h3><div class='circle' id='circle'></div><p>Tap the circle when it glows!</p><p id='resultMessage'></p>";
      const circle = document.getElementById("circle");
      let score = 0;
      circle.addEventListener("click", () => {
        score++;
        showResult("Nice rhythm! üí´ Score: " + score);
      });
    }

    // ‚òÅÔ∏è Catch the Clouds
    function loadCloudGame(container) {
      container.innerHTML = "<h3>‚òÅÔ∏è Catch the Clouds</h3><canvas id='cloudCanvas' width='400' height='300'></canvas><p id='resultMessage'></p>";
      const canvas = document.getElementById("cloudCanvas");
      const ctx = canvas.getContext("2d");
      let basketX = 180, score = 0, clouds = [];

      document.addEventListener("keydown", e => {
        if (e.key === "ArrowLeft") basketX -= 20;
        if (e.key === "ArrowRight") basketX += 20;
      });

      function createCloud() {
        clouds.push({ x: Math.random() * 370, y: 0 });
      }

      function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ctx.fillStyle = "#74c0fc";
        ctx.fillRect(basketX, 260, 40, 20);

        ctx.fillStyle = "#fff";
        for (let c of clouds) {
          ctx.beginPath();
          ctx.arc(c.x, c.y, 15, 0, Math.PI * 2);
          ctx.fill();
          c.y += 2;
          if (c.y > 260 && c.x > basketX && c.x < basketX + 40) {
            score++;
            clouds.splice(clouds.indexOf(c), 1);
            showResult("‚òÅÔ∏è Nice catch! Score: " + score);
          }
        }
      }

      setInterval(createCloud, 1000);
      setInterval(draw, 50);
    }

    function showResult(msg) {
      const res = document.getElementById("resultMessage");
      if (res) {
        res.textContent = msg;
        res.style.opacity = 1;
        setTimeout(() => (res.style.opacity = 0), 1500);
      }
    }
  </script>
